2. **WebSocket Server (Node example):**

   ```js
   import express from "express";
   import { WebSocketServer } from "ws";
   import fetch from "node-fetch";
   import { transcribeAudio, generateResponse, synthesizeVoice } from "./aiLogic.js";

   const app = express();
   const server = app.listen(3000, () => console.log("Listening on 3000"));
   const wss = new WebSocketServer({ noServer: true });

   server.on("upgrade", (req, socket, head) => {
     if (req.url === "/media-stream") {
       wss.handleUpgrade(req, socket, head, (ws) => handleMediaStream(ws));
     } else {
       socket.destroy();
     }
   });

   async function handleMediaStream(ws) {
     ws.on("message", async (message) => {
       const data = JSON.parse(message.toString());
       if (data.event === "media") {
         const audioBase64 = data.media.payload;
         const transcript = await transcribeAudio(audioBase64);
         const reply = await generateResponse(transcript);
         const audioReply = await synthesizeVoice(reply);
         ws.send(JSON.stringify({ event: "media", media: { payload: audioReply } }));
       }
     });
   }
   ```
